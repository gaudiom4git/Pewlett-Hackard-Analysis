Departments
-
dept_no varchar pk 
dept_name varchar

Dept_Emp
-
emp_no int pk
dept_no int fk - Departments.dept_no
from_date date
end_date date


Employees
-
emp_no int fk - Dept_Emp.emp_no
birth_date date
first_name varchar
last_name varchar
gender varchar
hire_date date


Managers
-
dept_no varchar fk - Departments.dept_no
emp_no int pk fk - Employees.emp_no 
from_date date
to_date date

Titles
-
emp_no int
title varchar
from_date date
to_date date

Salaries
-
emp_no int
salary float
from_date date
to_date date
